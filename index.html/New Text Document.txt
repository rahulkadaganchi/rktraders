    <!-- Interactive Cart / Basket Logic -->
    <script>
        let quoteBasket = [];

        function updateUI() {
            const listElement = document.getElementById('quote-list');
            const emptyNotice = document.getElementById('quote-basket-empty');
            const formElement = document.getElementById('quote-form');
            const counterElement = document.getElementById('cart-count');

            counterElement.innerText = quoteBasket.length;

            if (quoteBasket.length === 0) {
                emptyNotice.classList.remove('hidden');
                formElement.classList.add('hidden');
                listElement.innerHTML = '';
                return;
            }

            emptyNotice.classList.add('hidden');
            formElement.classList.remove('hidden');

            listElement.innerHTML = quoteBasket.map((item, index) => `
                <li class="flex items-center justify-between py-3 text-sm">
                    <span class="font-medium text-premiumNavy">${item.name}</span>
                    <div class="flex items-center gap-4">
                        <div class="flex items-center border border-gray-200 rounded">
                            <button onclick="changeQty(${index}, -1)" class="px-2 py-1 text-gray-500 hover:bg-gray-100">-</button>
                            <span class="px-3 font-semibold text-xs">${item.qty}</span>
                            <button onclick="changeQty(${index}, 1)" class="px-2 py-1 text-gray-500 hover:bg-gray-100">+</button>
                        </div>
                        <button onclick="removeItem(${index})" class="text-red-500 hover:text-red-700 text-xs uppercase tracking-wider font-semibold">Remove</button>
                    </div>
                </li>
            `).join('');
        }

        function addToQuote(itemName) {
            const existingIndex = quoteBasket.findIndex(item => item.name === itemName);
            if (existingIndex > -1) {
                quoteBasket[existingIndex].qty += 1;
            } else {
                quoteBasket.push({ name: itemName, qty: 1 });
            }
            updateUI();
            
            // Auto smooth scroll to quote summary window for dynamic elegant feedback
            document.getElementById('quote-section').scrollIntoView({ behavior: 'smooth' });
        }

        function changeQty(index, change) {
            quoteBasket[index].qty += change;
            if (quoteBasket[index].qty <= 0) {
                quoteBasket.splice(index, 1);
            }
            updateUI();
        }

        function removeItem(index) {
            quoteBasket.splice(index, 1);
            updateUI();
        }

        function submitQuoteForm(event) {
            event.preventDefault();
            alert("Thank you! Your quote request has been saved securely. Our sales representatives will map the pricing specs and forward your official PDF layout immediately.");
            quoteBasket = [];
            updateUI();
            event.target.reset();
        }
    </script>
